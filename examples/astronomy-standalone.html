astronomy.js 독립 사용법

별자리판 없이 순수 천문 계산 로직만 가져다 쓰는 방법

astronomy.js는 다른 라이브러리 의존성 없이 독립적으로 작동합니다.
필요한 계산 클래스만 import하여 본인의 프로젝트에 바로 사용하세요.

1. 율리우스일(JD) 및 시간 변환

import { AstroTime } from '../js/core/astronomy.js';

// 1. 특정 날짜의 율리우스일 계산
const jd = AstroTime.jd(2024, 12, 25, 21, 0, 0);

// 2. 서울(UTC+9, 경도 126.9) 기준 항성시 계산 객체 생성
const at = new AstroTime(9, 126.9, 37.5);
const lst = at.LCT2LST(jd); // 지방 항성시

2. 좌표 투영 (EquiDistanceProjection)

import { AstroMath, EquiDistanceProjection } from '../js/core/astronomy.js';

// 관측지 위도(서울) 기준 투영기 생성
const proj = new EquiDistanceProjection(300, 37.5 * AstroMath.D2R);

// 적경(RA) 6h, 적위(Dec) 45도인 별의 화면 x, y 좌표
const pos = proj.project(6 * AstroMath.H2R, 45 * AstroMath.D2R);
